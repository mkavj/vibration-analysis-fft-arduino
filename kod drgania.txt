#include <Wire.h>
#include <Adafruit_MPU6050.h>
#include <Adafruit_Sensor.h>
#include "arduinoFFT.h"

// --- KONFIGURACJA ---
// Dla Arduino Uno liczba próbek musi być mała (64), bo ma mało pamięci.
#define SAMPLES 64            
#define SAMPLING_FREQ 50       // Częstotliwość próbkowania w Hz

Adafruit_MPU6050 czujnik;
ArduinoFFT<double> FFT = ArduinoFFT<double>();

// Zmienne do FFT
double vReal[SAMPLES];
double vImag[SAMPLES];

// Zmienne do pozycjonowania (całkowanie)
float velocity_x = 0;
float position_x = 0;
unsigned long last_time = 0;
float calibration_offset_x = 0;

// Zmienne czasowe
unsigned int sampling_period_us;
unsigned long microseconds;

void setup() {
  Serial.begin(115200); // Pamiętaj ustawić tę prędkość w Monitorze!
  while (!Serial);

  // Uruchomienie czujnika
  if (!czujnik.begin()) {
    Serial.println("Błąd! Nie wykryto czujnika GY-521. Sprawdź kable (SDA->A4, SCL->A5).");
    while (1);
  }
 
  // Ustawienia czujnika (zakres i filtr)
  czujnik.setAccelerometerRange(MPU6050_RANGE_2_G);
  czujnik.setFilterBandwidth(MPU6050_BAND_44_HZ);

  sampling_period_us = round(1000000 * (1.0 / SAMPLING_FREQ));

  // --- KALIBRACJA (w spoczynku) ---
  Serial.println("Kalibracja... poloz czujnik plasko i nie ruszaj.");
  delay(1000);
  float sum = 0;
  for(int i=0; i<100; i++) {
    sensors_event_t a, g, temp;
    czujnik.getEvent(&a, &g, &temp);
    sum += a.acceleration.x;
    delay(10);
  }
  calibration_offset_x = sum / 100.0;
  Serial.print("Kalibracja zakonczona. Offset: "); Serial.println(calibration_offset_x);
  delay(1000);
 
  last_time = micros();
}

void loop() {
  // Pętla zbierająca 64 próbki
  for(int i = 0; i < SAMPLES; i++) {
    microseconds = micros();
   
    sensors_event_t a, g, temp;
    czujnik.getEvent(&a, &g, &temp);

    // --- CZĘŚĆ 1: Analiza w czasie (Wykres) ---
    unsigned long current_time = micros();
    float dt = (current_time - last_time) / 1000000.0;
    last_time = current_time;

    // Odczyt przyspieszenia bez grawitacji (offsetu)
    float accel_x = a.acceleration.x - calibration_offset_x;

    // Filtr szumów (martwa strefa) - żeby wykres nie odpływał sam z siebie
    if (abs(accel_x) < 0.15) {
      accel_x = 0;
    }

    // Całkowanie: Przyspieszenie -> Prędkość -> Pozycja
    velocity_x += accel_x * dt;
    position_x += velocity_x * dt;

    // DANE DO KREŚLARKI (SERIAL PLOTTER):
    Serial.print("AccX:"); Serial.print(accel_x);
    Serial.print("\tPosX:"); Serial.print(position_x);
    // Serial.println(); // Odkomentuj to, jeśli chcesz TYLKO wykres, bez tabelki FFT

    // Zbieranie danych do FFT
    vReal[i] = a.acceleration.x;
    vImag[i] = 0;

    // Czekanie na idealny czas próbkowania
    while(micros() - microseconds < sampling_period_us) {
      // czekaj...
    }
  }

  // --- CZĘŚĆ 2: Analiza FFT (Tabela) ---
  // Kod wchodzi tu raz na 64 próbki (czyli co ok. 1,5 sekundy)
 
  Serial.println(); // Nowa linia oddzielająca wykres od tabeli
  Serial.println("--- ANALIZA FFT (AMPLITUDA PRZEMIESZCZEN) ---");
 
  // Obliczenia FFT
  FFT.windowing(vReal, SAMPLES, FFT_WIN_TYP_HAMMING, FFT_FORWARD);
  FFT.compute(vReal, vImag, SAMPLES, FFT_FORWARD);
  FFT.complexToMagnitude(vReal, vImag, SAMPLES);

  // Wyświetlanie wyników
  for(int i = 2; i < (SAMPLES/2); i++) {
    double frequency = (i * 1.0 * SAMPLING_FREQ) / SAMPLES;
    double accel_amplitude = vReal[i];

    // Przeliczenie przyspieszenia na przemieszczenie (w mm)
    // Wzór: D = A / omega^2
    double omega = 2.0 * PI * frequency;
    double displacement_mm = 0;
   
    if (frequency > 1.0) {
        displacement_mm = (accel_amplitude / (omega * omega)) * 1000.0;
    }

    // Wyświetl tylko jeśli drgania są widoczne (>0.01 mm)
    if (displacement_mm > 0.01) {
      Serial.print("Freq: "); Serial.print(frequency, 1);
      Serial.print(" Hz \t -> Wychylenie: "); Serial.print(displacement_mm, 3);
      Serial.println(" mm");
    }
  }
  Serial.println("---------------------------------------------");
 
  // Opcjonalny reset pozycji co cykl (żeby wykres wracał do zera)
  // position_x = 0; velocity_x = 0;
}
